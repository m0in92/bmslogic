22 double calc_ocp_schmit(std::vector<double> &A_list, double &K, double &U0, double &SOC)
23 {
24 double R = Constants.R;
25 double F = Constants.F;
26 double temp = 298.15; // [K]
27
28 double sec_term = (R * temp / F) * std::log((1 - SOC) / SOC); // second term
29
30 // the following defines the third term
31 double pre_term = 1 / (std::pow(K * (2 * SOC - 1) + 1, 2));
32 double post_term = 0;
33 for (int i = 0; i < A_list.size(); i++)
34 {
35 post_term += (A_list[i] / F) * (std::pow(2 * SOC - 1, i + 1) - 2 * i * SOC * (1 - SOC) / std::pow(2 * SOC - 1, 1 - i));
36 }
37 double third_term = pre_term * post_term;
38
39 // the following defines the fourth term
40 post_term = 0;
41 for (int i = 0; i < A_list.size(); i++)
42 post_term += (A_list[i] / F) * (std::pow(2 * SOC - 1, i) * (2 * (i + 1) * (std::pow(SOC, 2)) - 2 * (i + 1) * SOC + 1));
43 double fou_term = K * post_term;
44
45 return U0 + sec_term + third_term + fou_term;
46 }
